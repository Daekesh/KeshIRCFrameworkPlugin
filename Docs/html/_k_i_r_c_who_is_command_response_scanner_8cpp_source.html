27 void UKIRCWhoIsCommandResponseScanner::HandleNumeric_Implementation( UKIRCUser* const Source, int32 Numeric, const TArray<FString>& Params, const FString& Message )
28 {
29  // Validity
30  if ( Client == NULL )
31  {
32  KIRCLog( Error, "Trying to check for whois command response with a null client." );
33  Fail( "Trying to check for whois command response with a null client." );
34  return;
35  }
36 
37  UKIRCServer* Server = Client->GetServer();
38 
39  if ( Server == NULL )
40  {
41  KIRCLog( Error, "Trying to check for whois command response with a null server." );
42  Fail( "Trying to check for whois command response with a null server." );
43  return;
44  }
45 
46  if ( Target.Len() == 0 )
47  {
48  KIRCLog( Error, "Trying to check for whois command response with a null target." );
49  Fail( "Trying to check for whois command response with a null target." );
50  return;
51  }
52 
53  if ( !Params[ 1 ].Equals( Target, ESearchCase::IgnoreCase ) )
54  return;
55 
56  // Errors
57  if ( Numeric == UKIRCClient::GetNumerics().ErrorNoSuchNick )
58  {
59  Super::HandleNumeric_Implementation( Source, Numeric, Params, Message );
60  return;
61  }
62 
63  // Replies
64  if ( Numeric == UKIRCClient::GetNumerics().ReplyWhoIsUser )
65  {
66  if ( Params.Num() < 4 )
67  {
68  KIRCLog( Error, "WhoIs user received without enough params." );
69  return;
70  }
71 
72  UserInfo.NickName = Params[ 1 ];
73  UserInfo.Ident = Params[ 2 ];
74  UserInfo.Host = Params[ 3 ];
75  UserInfo.RealName = Message;
76 
77  UKIRCUser* User = Server->GetUserByName( Params[ 1 ] );
78 
79  if ( User != NULL )
80  {
81  User->UpdateMask( Params[ 1 ] + "!" + Params[ 2 ] + "@" + Params[ 3 ] );
82  User->SetRealName( Message );
83  }
84  }
85 
86  else if ( Numeric == UKIRCClient::GetNumerics().ReplyWhoIsChannels )
87  {
88  TArray<FString> ChannelList;
89  Message.ParseIntoArray( ChannelList, L" " );
90 
91  for ( const FString& Channel : ChannelList )
92  {
93  FKIRCWhoIsResponseChannelInfo ChannelInfo;
94 
95  for ( int32 i = 0; i < Channel.Len(); ++i )
96  {
97  if ( Channel[ i ] == UKIRCClient::GetModes().Channel.OpsPrefix[ 0 ] )
98  {
99  ChannelInfo.bOp = true;
100  continue;
101  }
102 
103  if ( Channel[ i ] == UKIRCClient::GetModes().Channel.HalfOpsPrefix[ 0 ] )
104  {
105  ChannelInfo.bHalfOp = true;
106  continue;
107  }
108 
109  if ( Channel[ i ] == UKIRCClient::GetModes().Channel.VoicePrefix[ 0 ] )
110  {
111  ChannelInfo.bVoice = true;
112  continue;
113  }
114 
115  ChannelInfo.Channel = Channel.Mid( i );
116  UserInfo.Channels.Add( ChannelInfo );
117  break;
118  }
119  }
120  }
121 
122  else if ( Numeric == UKIRCClient::GetNumerics().ReplyWhoIsServer )
123  {
124  if ( Params.Num() < 3 )
125  {
126  KIRCLog( Error, "Received whois server reply without a server address." );
127  return;
128  }
129 
130  UserInfo.ServerHost = Params[ 2 ];
131  UserInfo.ServerName = Message;
132  }
133 
134  else if ( Numeric == UKIRCClient::GetNumerics().ReplyWhoIsOperator )
135  UserInfo.bOperator = true;
136 
137  else if ( Numeric == UKIRCClient::GetNumerics().ReplyAway )
138  {
139  UserInfo.bAway = true;
140  UserInfo.AwayMessage = Message;
141  }
142 
143  else if ( Numeric == UKIRCClient::GetNumerics().ReplyWhoIsIdle )
144  {
145  if ( Params.Num() < 3 )
146  {
147  KIRCLog( Error, "Received whois idle reply without an idle time." );
148  return;
149  }
150 
151  UserInfo.bIdle = true;
152  UserInfo.SecondsIdle = FCString::Atoi( *Params[ 2 ] );
153  }
154 
155  else if ( Numeric == UKIRCClient::GetNumerics().ReplyWhoAuthName )
156  {
157  if ( Params.Num() < 3 )
158  {
159  KIRCLog( Error, "Received whois auth name reply without an auth name." );
160  return;
161  }
162 
163  UserInfo.AuthName = Params[ 2 ];
164  }
165 
166  else if ( Numeric == UKIRCClient::GetNumerics().ReplyEndOfWhoIs )
167  {
168  Success();
169  }
170 }
